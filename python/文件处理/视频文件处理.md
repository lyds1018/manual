------

### (1) 读取视频

```
import cv2

cap = cv2.VideoCapture("video.mp4")  # 打开视频文件

while cap.isOpened():
    ret, frame = cap.read()          # 逐帧读取
    if not ret:
        break
    cv2.imshow("frame", frame)       # 显示
    if cv2.waitKey(25) & 0xFF == ord('q'):  # 按 q 退出
        break

cap.release()
cv2.destroyAllWindows()
```

------

### (2) 获取视频信息

```
fps = cap.get(cv2.CAP_PROP_FPS)              # 帧率
width = int(cap.get(cv2.CAP_PROP_FRAME_WIDTH))
height = int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT))
frame_count = int(cap.get(cv2.CAP_PROP_FRAME_COUNT))
print(f"分辨率: {width}x{height}, 帧率: {fps}, 总帧数: {frame_count}")
```

------

### (3) 保存视频（写入）

```
fourcc = cv2.VideoWriter_fourcc(*"mp4v")   # 编码格式 (mp4v, XVID, etc.)
out = cv2.VideoWriter("output.mp4", fourcc, fps, (width, height))

cap = cv2.VideoCapture("video.mp4")
while cap.isOpened():
    ret, frame = cap.read()
    if not ret:
        break
    out.write(frame)   # 保存帧
cap.release()
out.release()
```
